## 3. Relational Schema Specification (MySQL DDL)

```sql
SET FOREIGN_KEY_CHECKS = 0;

-- ============================================================================
-- SYSTEM TYPE LOOKUP DICTIONARIES
-- ============================================================================

CREATE TABLE PARTY_TYPE (
    PARTY_TYPE_ID VARCHAR(40) NOT NULL,
    DESCRIPTION VARCHAR(255) NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (PARTY_TYPE_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE PARTY_IDENTIFICATION_TYPE (
    PARTY_IDENTIFICATION_TYPE_ID VARCHAR(40) NOT NULL,
    DESCRIPTION VARCHAR(255) NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (PARTY_IDENTIFICATION_TYPE_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE CONTACT_MECH_TYPE (
    CONTACT_MECH_TYPE_ID VARCHAR(40) NOT NULL,
    DESCRIPTION VARCHAR(255) NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (CONTACT_MECH_TYPE_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE BEHAVIORAL_EVENT_TYPE (
    BEHAVIORAL_EVENT_TYPE_ID VARCHAR(40) NOT NULL,
    DESCRIPTION VARCHAR(255) NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (BEHAVIORAL_EVENT_TYPE_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ============================================================================
-- MASTER IDENTITY DOMAIN TABLES
-- ============================================================================

CREATE TABLE PARTY (
    PARTY_ID VARCHAR(40) NOT NULL,
    PARTY_TYPE_ID VARCHAR(40) NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (PARTY_ID),
    CONSTRAINT FK_PARTY_TYPE FOREIGN KEY (PARTY_TYPE_ID) REFERENCES PARTY_TYPE (PARTY_TYPE_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE PERSON (
    PARTY_ID VARCHAR(40) NOT NULL,
    FIRST_NAME VARCHAR(100) NULL,
    LAST_NAME VARCHAR(100) NULL,
    BIRTH_DATE DATE NULL,
    GENDER VARCHAR(20) NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (PARTY_ID),
    CONSTRAINT FK_PERSON_PARTY FOREIGN KEY (PARTY_ID) REFERENCES PARTY (PARTY_ID) ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE ORGANIZATION (
    PARTY_ID VARCHAR(40) NOT NULL,
    ORGANIZATION_NAME VARCHAR(255) NOT NULL,
    REGISTRATION_NUMBER VARCHAR(100) NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (PARTY_ID),
    CONSTRAINT FK_ORG_PARTY FOREIGN KEY (PARTY_ID) REFERENCES PARTY (PARTY_ID) ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE PARTY_IDENTIFICATION (
    PARTY_ID VARCHAR(40) NOT NULL,
    PARTY_IDENTIFICATION_TYPE_ID VARCHAR(40) NOT NULL,
    ID_VALUE VARCHAR(255) NOT NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (PARTY_ID, PARTY_IDENTIFICATION_TYPE_ID),
    CONSTRAINT FK_PID_PARTY FOREIGN KEY (PARTY_ID) REFERENCES PARTY (PARTY_ID),
    CONSTRAINT FK_PID_TYPE FOREIGN KEY (PARTY_IDENTIFICATION_TYPE_ID) REFERENCES PARTY_IDENTIFICATION_TYPE (PARTY_IDENTIFICATION_TYPE_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE INDEX IDX_PID_LOOKUP ON PARTY_IDENTIFICATION (PARTY_IDENTIFICATION_TYPE_ID, ID_VALUE);

-- ============================================================================
-- SYSTEM INFRASTRUCTURE QUEUES & LOGS
-- ============================================================================

CREATE TABLE SYSTEM_MESSAGE_TYPE (
    SYSTEM_MESSAGE_TYPE_ID VARCHAR(40) NOT NULL,
    DESCRIPTION VARCHAR(255) NULL,
    CONSUME_SERVICE_NAME VARCHAR(255) NULL,
    CONTENT_TYPE VARCHAR(100) NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (SYSTEM_MESSAGE_TYPE_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE SYSTEM_MESSAGE_REMOTE (
    SYSTEM_MESSAGE_REMOTE_ID VARCHAR(40) NOT NULL,
    DESCRIPTION VARCHAR(255) NULL,
    SHARED_SECRET VARCHAR(255) NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (SYSTEM_MESSAGE_REMOTE_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE SYSTEM_MESSAGE (
    SYSTEM_MESSAGE_ID VARCHAR(40) NOT NULL,
    SYSTEM_MESSAGE_TYPE_ID VARCHAR(40) NULL,
    SYSTEM_MESSAGE_REMOTE_ID VARCHAR(40) NULL,
    STATUS_ID VARCHAR(40) NOT NULL,
    MESSAGE_TEXT LONGTEXT NULL,
    INIT_DATE DATETIME NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (SYSTEM_MESSAGE_ID),
    CONSTRAINT FK_SM_TYPE FOREIGN KEY (SYSTEM_MESSAGE_TYPE_ID) REFERENCES SYSTEM_MESSAGE_TYPE (SYSTEM_MESSAGE_TYPE_ID),
    CONSTRAINT FK_SM_REMOTE FOREIGN KEY (SYSTEM_MESSAGE_REMOTE_ID) REFERENCES SYSTEM_MESSAGE_REMOTE (SYSTEM_MESSAGE_REMOTE_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE INDEX IDX_SM_POLLING ON SYSTEM_MESSAGE (SYSTEM_MESSAGE_TYPE_ID, STATUS_ID);

CREATE TABLE SYSTEM_MESSAGE_ERROR (
    SYSTEM_MESSAGE_ID VARCHAR(40) NOT NULL,
    ERROR_SEQUENCE_ID VARCHAR(40) NOT NULL,
    ERROR_TEXT LONGTEXT NULL,
    ERROR_DATE DATETIME NULL,
    PRIMARY KEY (SYSTEM_MESSAGE_ID, ERROR_SEQUENCE_ID),
    CONSTRAINT FK_SME_MSG FOREIGN KEY (SYSTEM_MESSAGE_ID) REFERENCES SYSTEM_MESSAGE (SYSTEM_MESSAGE_ID) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE DATA_MANAGER_CONFIG (
    DATA_MANAGER_CONFIG_ID VARCHAR(40) NOT NULL,
    DESCRIPTION VARCHAR(255) NULL,
    IMPORT_SERVICE_NAME VARCHAR(255) NOT NULL,
    EXECUTION_MODE_ID VARCHAR(40) NULL,
    MULTI_THREADING CHAR(1) NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (DATA_MANAGER_CONFIG_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE DATA_MANAGER_LOG (
    LOG_ID VARCHAR(40) NOT NULL,
    DATA_MANAGER_CONFIG_ID VARCHAR(40) NULL,
    STATUS_ID VARCHAR(40) NOT NULL,
    TOTAL_RECORD_COUNT BIGINT NULL,
    FAILED_RECORD_COUNT BIGINT NULL,
    FROM_DATE DATETIME NULL,
    THRU_DATE DATETIME NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (LOG_ID),
    CONSTRAINT FK_DML_CONFIG FOREIGN KEY (DATA_MANAGER_CONFIG_ID) REFERENCES DATA_MANAGER_CONFIG (DATA_MANAGER_CONFIG_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE DATA_MANAGER_CONTENT (
    LOG_ID VARCHAR(40) NOT NULL,
    CONTENT_TYPE_ID VARCHAR(40) NOT NULL,
    FILE_PATH VARCHAR(255) NOT NULL,
    PRIMARY KEY (LOG_ID, CONTENT_TYPE_ID),
    CONSTRAINT FK_DMC_LOG FOREIGN KEY (LOG_ID) REFERENCES DATA_MANAGER_LOG (LOG_ID) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ============================================================================
-- ROLES & UNIFIED CONTACT MECHANISMS
-- ============================================================================

CREATE TABLE ROLE_TYPE (
    ROLE_TYPE_ID VARCHAR(40) NOT NULL,
    DESCRIPTION VARCHAR(255) NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (ROLE_TYPE_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE PARTY_ROLE (
    PARTY_ID VARCHAR(40) NOT NULL,
    ROLE_TYPE_ID VARCHAR(40) NOT NULL,
    FROM_DATE DATETIME NOT NULL,
    THRU_DATE DATETIME NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (PARTY_ID, ROLE_TYPE_ID, FROM_DATE),
    CONSTRAINT FK_PR_PARTY FOREIGN KEY (PARTY_ID) REFERENCES PARTY (PARTY_ID),
    CONSTRAINT FK_PR_TYPE FOREIGN KEY (ROLE_TYPE_ID) REFERENCES ROLE_TYPE (ROLE_TYPE_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE CONTACT_MECHANISM (
    CONTACT_MECH_ID VARCHAR(40) NOT NULL,
    CONTACT_MECH_TYPE_ID VARCHAR(40) NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (CONTACT_MECH_ID),
    CONSTRAINT FK_CM_TYPE FOREIGN KEY (CONTACT_MECH_TYPE_ID) REFERENCES CONTACT_MECH_TYPE (CONTACT_MECH_TYPE_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE EMAIL_ADDRESS (
    CONTACT_MECH_ID VARCHAR(40) NOT NULL,
    EMAIL_ADDRESS VARCHAR(255) NOT NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (CONTACT_MECH_ID),
    CONSTRAINT FK_EA_CM FOREIGN KEY (CONTACT_MECH_ID) REFERENCES CONTACT_MECHANISM (CONTACT_MECH_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE UNIQUE INDEX UNIQUE_EMAIL_STR ON EMAIL_ADDRESS (EMAIL_ADDRESS);

CREATE TABLE TELECOM_NUMBER (
    CONTACT_MECH_ID VARCHAR(40) NOT NULL,
    COUNTRY_CODE VARCHAR(10) NULL,
    AREA_CODE VARCHAR(10) NULL,
    CONTACT_NUMBER VARCHAR(30) NOT NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (CONTACT_MECH_ID),
    CONSTRAINT FK_TN_CM FOREIGN KEY (CONTACT_MECH_ID) REFERENCES CONTACT_MECHANISM (CONTACT_MECH_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE POSTAL_ADDRESS (
    CONTACT_MECH_ID VARCHAR(40) NOT NULL,
    TO_NAME VARCHAR(255) NULL,
    ADDRESS1 VARCHAR(255) NOT NULL,
    ADDRESS2 VARCHAR(255) NULL,
    CITY VARCHAR(100) NOT NULL,
    STATE_PROVINCE_GEO_ID VARCHAR(40) NULL,
    POSTAL_CODE VARCHAR(30) NULL,
    COUNTRY_GEO_ID VARCHAR(40) NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (CONTACT_MECH_ID),
    CONSTRAINT FK_PA_CM FOREIGN KEY (CONTACT_MECH_ID) REFERENCES CONTACT_MECHANISM (CONTACT_MECH_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE PARTY_CONTACT_MECH (
    PARTY_ID VARCHAR(40) NOT NULL,
    CONTACT_MECH_ID VARCHAR(40) NOT NULL,
    FROM_DATE DATETIME NOT NULL,
    THRU_DATE DATETIME NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (PARTY_ID, CONTACT_MECH_ID, FROM_DATE),
    CONSTRAINT FK_PCM_PARTY FOREIGN KEY (PARTY_ID) REFERENCES PARTY (PARTY_ID),
    CONSTRAINT FK_PCM_CM FOREIGN KEY (CONTACT_MECH_ID) REFERENCES CONTACT_MECHANISM (CONTACT_MECH_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ============================================================================
-- TIME-SERIES TIMELINES & COHORT SEGMENTATION
-- ============================================================================

CREATE TABLE BEHAVIORAL_EVENT (
    EVENT_ID VARCHAR(40) NOT NULL,
    PARTY_ID VARCHAR(40) NULL,
    BEHAVIORAL_EVENT_TYPE_ID VARCHAR(40) NOT NULL,
    EVENT_DATE DATETIME NOT NULL,
    TARGET_PRODUCT_ID VARCHAR(40) NULL,
    EVENT_VALUE_NUMERIC DECIMAL(24,4) DEFAULT 0,
    EVENT_PAYLOAD LONGTEXT NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (EVENT_ID),
    CONSTRAINT FK_BE_PARTY FOREIGN KEY (PARTY_ID) REFERENCES PARTY (PARTY_ID),
    CONSTRAINT FK_BE_TYPE FOREIGN KEY (BEHAVIORAL_EVENT_TYPE_ID) REFERENCES BEHAVIORAL_EVENT_TYPE (BEHAVIORAL_EVENT_TYPE_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE INDEX IDX_BE_PERFORMANCE ON BEHAVIORAL_EVENT (PARTY_ID, BEHAVIORAL_EVENT_TYPE_ID, EVENT_DATE);
CREATE INDEX IDX_BE_FAST_SEGMENT ON BEHAVIORAL_EVENT (BEHAVIORAL_EVENT_TYPE_ID, TARGET_PRODUCT_ID, EVENT_VALUE_NUMERIC);

CREATE TABLE SEGMENT (
    SEGMENT_ID VARCHAR(40) NOT NULL,
    SEGMENT_NAME VARCHAR(255) NOT NULL,
    SEGMENT_RULE_CRITERIA LONGTEXT NOT NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (SEGMENT_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE SEGMENT_MEMBERSHIP (
    SEGMENT_ID VARCHAR(40) NOT NULL,
    PARTY_ID VARCHAR(40) NOT NULL,
    FROM_DATE DATETIME NOT NULL,
    THRU_DATE DATETIME NULL,
    LAST_UPDATED_STAMP DATETIME NULL,
    TX_STAMP DATETIME NULL,
    PRIMARY KEY (SEGMENT_ID, PARTY_ID, FROM_DATE),
    CONSTRAINT FK_SM_SEGMENT FOREIGN KEY (SEGMENT_ID) REFERENCES SEGMENT (SEGMENT_ID),
    CONSTRAINT FK_SM_PARTY FOREIGN KEY (PARTY_ID) REFERENCES PARTY (PARTY_ID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE INDEX IDX_SM_AUDITING ON SEGMENT_MEMBERSHIP (SEGMENT_ID, THRU_DATE);

SET FOREIGN_KEY_CHECKS = 1;
